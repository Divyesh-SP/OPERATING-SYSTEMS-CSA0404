Experiment 21-UNIX File Management using System Calls

#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>

int main() {
    int fd;
    char buffer[100];

    fd = open("sample.txt", O_CREAT | O_WRONLY, 0644);
    write(fd, "Hello UNIX File System", 22);
    close(fd);

    fd = open("sample.txt", O_RDONLY);
    read(fd, buffer, 22);
    buffer[22] = '\0';

    printf("File Content: %s\n", buffer);
    close(fd);

    return 0;
}
